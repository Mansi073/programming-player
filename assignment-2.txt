Add `int max_of_four(int a, int b, int c, int d)` here.
*/
int max_of_four(int a,int b,int c,int d){
    if (a>b && a>c && a>d){
        return a;
    }
    else if (b>c&&b>d){
        return b;
    }
    else if (c>d){
        return c;
    }
    else{
        return d;
    }
    
}
    

int main() {
    int a, b, c, d;
    scanf("%d %d %d %d", &a, &b, &c, &d);
    int ans = max_of_four(a, b, c, d);
    printf("%d", ans);
    
    return 0;
}

#include <stdio.h>

void update(int *a,int *b) {
       int c,d;
       c=(*a + *b);
       d=abs(*b - *a);
       *a=c;
       *b=d;
}

int main() {
    int a, b;
    int *pa = &a, *pb = &b;
    
    scanf("%d %d", &a, &b);
    update(pa, pb);
    printf("%d\n%d", a, b);

    return 0;
}
